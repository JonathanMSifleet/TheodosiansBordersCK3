on_game_start_after_lobby = {
	effect = {
		# If h_roman_empire already exists at game start (e.g. a bookmark where it was restored),
		# run the full setup. Otherwise just set de jure structure for the sub-empires.
		if = {
			limit = { exists = title:h_roman_empire.holder }
			title:h_roman_empire.holder = { ewrh_create_sub_empires_effect = yes }
		}
		else = {
			# Set de jure hegemony/empire structure without needing a holder.
			# Only core Italian/Croatian titles are assigned to e_western_roman_empire.
			# e_byzantium vassal structure is preserved by the destroy loop exclusion
			# in create_roman_empire_scripted_effect, so no de jure needed here either.
			title:e_western_roman_empire = { set_de_jure_liege_title = title:h_roman_empire }
			title:e_byzantium = { set_de_jure_liege_title = title:h_roman_empire }

			title:k_italy = { set_de_jure_liege_title = title:e_western_roman_empire }
			title:k_romagna = { set_de_jure_liege_title = title:e_western_roman_empire }
			title:k_naples = { set_de_jure_liege_title = title:e_western_roman_empire }
			title:k_sardinia = { set_de_jure_liege_title = title:e_western_roman_empire }
			title:k_sicily = { set_de_jure_liege_title = title:e_western_roman_empire }
			title:k_croatia = { set_de_jure_liege_title = title:e_western_roman_empire }
		}
	}
}
